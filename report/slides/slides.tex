% ===== 3. Learned policies & why they're brittle =====
\begin{frame}{The robot is driven by a learned policy --- a brittle one}
  \begin{itemize}
    \item \hl{Imitation learning}: a neural policy trained to mimic \hl{demonstrations}.
    \item \hl{VLA} (vision--language--action): cameras $+$ joint state $+$ a \hl{language instruction} $\rightarrow$ actions. Here: \hl{SmolVLA}.
    \item Trained on \hl{successful} demos only $\Rightarrow$ any slip lands it \hl{out of distribution}; errors \hl{compound} over the horizon.
    \item Nothing \hl{detects} or \hl{interrupts} a failure mid-episode.
  \end{itemize}
  \vfill
  \begin{block}{Angle}
    Fix the \hl{structure around the policy}, not just the policy.
  \end{block}
\end{frame}

% ===== 4. Idea + contributions =====
\begin{frame}{Idea: decompose, then orchestrate}
  \textbf{Hypothesis.} The task becomes more robust if split into \hl{simple language-instructed steps}, with a controller that \hl{checks each step's outcome} and \hl{recovers} from failure.
  \vfill
  \textbf{Contributions.}
  \begin{enumerate}
    \item \hl{End-to-end baselines} (ACT vs.\ SmolVLA): a stronger policy class alone is \hl{not enough}.
    \item A custom \hl{teleoperated step dataset} $+$ an \hl{orchestrator} that retries or switches target.
    \item \hl{Autonomous data generation}: does more data help? (\emph{not clear-cut})
  \end{enumerate}
\end{frame}

% ===== 6. Two formulations =====
\begin{frame}{Two formulations of the same task}
  \begin{itemize}
    \item \hl{Monotask}: one fixed prompt, \hl{end-to-end} --- no visibility into progress.
    \item \hl{Subtask}: decompose into \hl{GRASP $\rightarrow$ LIFT $\rightarrow$ PLACE} --- each step a \hl{checkpoint} whose \hl{physical outcome is verified}.
    \item GRASP names its target \hl{by position}: ``grasp the \emph{left / middle / right} orange''.
    \item \hl{Confirmed-grasp gate}: LIFT starts only on a \hl{secure} grasp (centred $+$ firm contact force) --- not just a closed gripper.
  \end{itemize}
  \vfill
  \begin{columns}[c]
    \column{0.32\textwidth}\centering\includegraphics[height=2.4cm]{grasp.png}\\\textbf{GRASP}
    \column{0.32\textwidth}\centering\includegraphics[height=2.4cm]{pick.png}\\\textbf{LIFT}
    \column{0.32\textwidth}\centering\includegraphics[height=2.4cm]{place.png}\\\textbf{PLACE}
  \end{columns}
\end{frame}

% ===== 7. Orchestrator =====
\begin{frame}{The orchestrator: a controller around the VLA}
  \begin{columns}[c]
    \begin{column}{0.54\textwidth}
      \begin{itemize}
        \item Checks read \hl{privileged simulator state} (poses, contact forces) --- the \hl{policy never sees it}.
        \item \hl{Timeout} = per-subtask step budget $\rightarrow$ the detection trigger.
        \item \hl{Level-1 retry}: slip / timeout $\rightarrow$ re-GRASP the \hl{same} orange.
        \item \hl{Level-2 redirect}: repeated GRASP failure $\rightarrow$ a \hl{different} orange --- the task keeps scoring.
        \item \hl{Home reset} (precondition): the policy only switches target \hl{far from all oranges}.
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \centering
      \includegraphics[width=\textwidth]{orchestrator_decision_flow.pdf}
    \end{column}
  \end{columns}
\end{frame}

% ===== 10. Results: grasp bottleneck =====
\begin{frame}{Grasp acquisition is the bottleneck}
  \begin{itemize}
    \item GRASP secures an orange in only \hl{$\sim$37\%} of attempts $\rightarrow$ \gn{55\%} under \hl{full} tuning.
    \item \hl{LIFT and PLACE already reliable} across every variant.
    \item Monotask (no grasp gate): lift completion \hl{56\% $\rightarrow$ 94\%} with full tuning.
  \end{itemize}

% ===== 11. Results: recovery at matched stalls =====
\begin{frame}{Recovery, compared at matched stalls}
  {\footnotesize A \hl{stall} = one grasp budget stuck on a single orange (monotask intent read from \emph{proximity}). What happens next?}
  \vfill
  \centering\small
  \begin{tabular}{llccc}
    \toprule
    Training & Form. & Redirected & Placed after & \dots via \emph{different} orange \\
    \midrule
    Teleop      & Monotask          & 53.5\%     & 31.4\%        & 16.3\% \\
    Teleop      & \textbf{Subtask}  & \gn{100\%} & \gn{71.2\%}   & \gn{53.4\%} \\
    \midrule
    Teleop+Auto & Monotask          & 45.2\%     & 34.4\%        & 17.2\% \\
    Teleop+Auto & \textbf{Subtask}  & \gn{100\%} & \gn{43.2\%}   & \gn{33.7\%} \\
    \bottomrule
  \end{tabular}
  \vfill
  \begin{itemize}
    \item Orchestrator \hl{redirects every time} (Level-2 by design); monotask switches in \hl{$\sim$half}.
    \item The gap sits \hl{entirely in the different-orange column}; Level-1 stays marginal.
  \end{itemize}
\end{frame}

% ===== 13. Takeaways + future =====
\begin{frame}{Takeaways \& future work}
  \begin{itemize}
    \item Orchestration helps \hl{most where grasping is weak}: almost never ends empty-handed (Level-2 redirection).
    \item It \hl{cannot fix a weak grasp} --- a \hl{stronger policy} did (full fine-tuning, vision), not more structure.
    \item Two \hl{complementary levers}: better grasping $+$ graceful failure.
    \item This task shows the orchestration's \hl{floor}, not its ceiling.
  \end{itemize}
  \vfill
  \textbf{Next:} RL / targeted grasp demos \; $\cdot$ \; \hl{from-any-state} data \; $\cdot$ \; \hl{VLM success monitor}.
\end{frame}

% ----- B3: limitations -----
\begin{frame}{Backup: limitations}
  \begin{itemize}
    \item \hl{Privileged supervision} --- success read from simulator state; deployment needs perception.
    \item \hl{Matched-regime confound} --- full variant trains at batch 32 / 40k steps (controls bound it).
    \item \hl{Composite-label inconsistency} --- teleop labels by eye vs.\ scripted 3\,cm rule.
    \item \hl{Limited sampling} --- 100 seeded episodes per model.
    \item \hl{Slow execution} --- deliberately slow teleop $\rightarrow$ long rollouts.
  \end{itemize}
\end{frame}

% ===== 3. Idea =====
\begin{frame}{The idea: separate the doing from the thinking}
  \begin{itemize}
    \item Split \hl{doing} from \hl{thinking} --- \hl{low level} $=$ execute one \hl{atomistic subtask}; \hl{high level} $=$ decide \emph{which} subtask, \emph{when}
    \item \hl{Low level} $=$ the \hl{VLA} (vision--language--action) --- a reflexive \hl{executor}, one subtask at a time \; ($\Rightarrow$ \hl{should be good enough})
    \item \hl{High level} $=$ a \hl{controller} --- \emph{what} to do / \emph{when} to recover; upgradeable on its own (\hl{algorithm} now $\rightarrow$ \hl{stronger model} later)
    \item \hl{Language} $=$ the \hl{bridge} --- interface between the levels
  \end{itemize}

% ===== 13. Conclusion & limitations =====
\begin{frame}{Conclusion \& limitations}
  \begin{columns}[t]
    \begin{column}{0.52\textwidth}
      \small
      \textbf{Conclusion}
      \begin{itemize}
        \item \hl{Strongest on weak policies} --- eliminates \hl{complete failure}: rarely ends \hl{empty-handed} (abandon $+$ place another)
        \item \hl{Auto-data doesn't seem useful} --- self-generated subtasks gave \hl{no reliable gain}
        \item \hl{Shows its floor here} --- a simple task (interchangeable objects, ordering-free); \hl{richer tasks} $\rightarrow$ bigger payoff
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \small
      \textbf{Limitations}
      \begin{itemize}
        \item \hl{Privileged supervision} --- sim state; deployment needs perception
        \item \hl{Not a fair comparison} --- the full models trained with different settings (batch / steps / chunk); controls bound it
        \item \hl{Chronological data only} --- obeys prompt $\sim$2/3; can't redirect near an orange $\rightarrow$ home reset
        \item \hl{Too-simple ``RL''} --- self-improvement loop with \hl{no real exploration}; reward $=$ just a \hl{success bit}
        \item \hl{Few, noisy evaluations} --- small episode count $+$ \hl{non-deterministic} sim $\rightarrow$ small gaps are noise
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% ===== 14. Future work =====
\begin{frame}{Future work}
  \textbf{Subtask execution} \; --- \; the binding constraint (\hl{grasp acquisition}), largely a \hl{data} matter
  \begin{itemize}
    \item \hl{Better data} --- a \hl{6-DOF arm} (SO-101's 5 joints can't track a full hand pose; Quest 3 pipeline failed) $\rightarrow$ larger, cleaner dataset \; $\cdot$ \; \hl{RL / targeted demos} for the hard grasps
    \item \hl{From-any-state} data --- grasp from out-of-sequence states $\rightarrow$ obey the prompt anywhere, \hl{redirect without the home reset}
  \end{itemize}
  \vfill
  \textbf{Orchestration} \; --- \; largely orthogonal to the policy, generalise it
  \begin{itemize}
    \item \hl{Drop the privileged check} --- judge subtask completion from \hl{observations}: the \hl{VLA itself} signals done, or an \hl{external model} $\rightarrow$ deployable on a \hl{real robot}
    \item \hl{Richer tasks} --- ordering constraints, irreversible steps $\rightarrow$ an \hl{FSM or LM/VLM} selects subtasks; VLA signals completion
  \end{itemize}
\end{frame}

% ===== 2. The problem =====
\begin{frame}{Long-horizon manipulation: failure is the hard part}
  \begin{itemize}
    \item A pick-and-place task succeeds only if the robot \hl{grasps $\rightarrow$ lifts $\rightarrow$ places} --- in that order.
    \item The hard problem is not a single failed step, but \hl{what the system does \emph{after} a failure}.
    \item \textbf{Stagnation:} a policy can keep retrying one orange until the timeout, ignoring the others.
    \item \textbf{State poisoning (hypothesis):} a fumbled grasp can knock an orange into a rarely-seen state, from which the policy acts unreliably.
  \end{itemize}
  \vfill
  \begin{block}{The angle}
    Improve robustness by \hl{fixing the structure around the policy}, not just the policy itself.
  \end{block}
\end{frame}

% ===== 3. Why learned policies are brittle =====
\begin{frame}{Why learned policies are brittle here}
  \begin{itemize}
    \item Trained on \hl{successful demonstrations} --- the training distribution is mostly ``things went well so far''.
    \item During autonomous execution, small deviations $\rightarrow$ \hl{out-of-distribution states} $\rightarrow$ errors accumulate over the long horizon.
    \item Vision-Language-Action (VLA) models add language as an interface --- e.g.\ \emph{``place the object in the plate''}.
    \item But that instruction \hl{assumes the object is already held}; language alone offers no way to \emph{detect} or \emph{interrupt} a failure.
  \end{itemize}
\end{frame}

% ===== 4. Hypothesis & contributions =====
\begin{frame}{Idea: decompose, then orchestrate}
  \textbf{Hypothesis.} Long-horizon tasks become more robust if decomposed into \hl{language-conditioned subtasks} surrounded by an \hl{orchestrator} that checks each outcome and recovers from failure.
  \vfill
  \textbf{Contributions.}
  \begin{enumerate}
    \item A monolithic baseline (\hl{ACT vs.\ SmolVLA}) --- changing the policy class alone is not enough.
    \item A targeted \hl{Teleop subtask dataset} + an orchestrator that monitors outcomes and performs \hl{retry / reset / redirect}.
    \item Using the orchestrator for \hl{autonomous data generation} --- and asking whether more data helps (a useful \emph{negative} result).
  \end{enumerate}
\end{frame}

% ===== 6. SmolVLA & fine-tuning =====
\begin{frame}{The policy: SmolVLA, fine-tuned efficiently}
  \begin{itemize}
    \item Pretrained vision-language backbone (\texttt{SmolVLM2-500M}) + a small \hl{action expert} (flow matching).
    \item Outputs a \hl{chunk of 20 joint actions} per query (matches the 20-frame terminal hold used in data collection).
    \item \hl{Parameter-efficient fine-tuning}: freeze the vision encoder \emph{and} language backbone; train only the \hl{action expert + state projection} ($\approx$100M of 450M params, $\sim$22\%).
    \item Keeps the web-scale visual/linguistic priors; adapts the action pathway to the SO-101.
  \end{itemize}
\end{frame}

% ===== 7. Baseline (defines "monotask" + 41% ceiling) =====
\begin{frame}{Baseline: is the bottleneck the policy class?}
  \begin{columns}[T]

% ===== 8. Subtask decomposition =====
\begin{frame}{Decompose: GRASP $\rightarrow$ LIFT $\rightarrow$ PLACE}
  \begin{itemize}
    \item Reformulate the full task as a sequence of subtasks --- each an explicit \hl{checkpoint}.
    \item Verify the \hl{physical outcome} before committing to the next step.
    \item GRASP targets \hl{one orange by relative position} (left / middle / right).
  \end{itemize}

  \begin{columns}[T]

        \item 1 episode = \hl{1 successful subtask}, labelled with its instruction.
        \item \hl{Randomised pick order} $\rightarrow$ forces grounding of the position instruction (target $\neq$ task progress).
        \item \hl{``Freeze tail''} (20 frames) $\rightarrow$ the policy learns to \emph{wait} at a boundary instead of drifting.
        \item Trained SmolVLA as a \hl{subtask executor} (GRASP / LIFT / PLACE).
      \end{itemize}

% ===== 10. Key insight =====
\begin{frame}{Key insight: language is spatially context-dependent}
  \begin{block}{The behavioural finding that motivates the orchestrator's reset}
  \begin{itemize}
    \item Gripper near orange A $\Rightarrow$ ``grasp the \emph{right} orange'' is \hl{ignored}.
    \item Target switching only works when the arm is \hl{far from all oranges}.
    \item[$\Rightarrow$] Before any target change, the arm must \hl{first move away}.
  \end{itemize}
  \end{block}
  \vspace{0.4em}
  This is exactly why the orchestrator needs a \hl{scripted home-pose reset} --- defined next.
\end{frame}

% ===== 11. Failures vs timeouts =====
\begin{frame}{Two things to detect: failures vs.\ timeouts}
  \textbf{Physical failure} --- the orange is actually lost:
  \begin{itemize}
    \item \textbf{LIFT} --- slips before reaching the plate.\quad \textbf{PLACE} --- dropped outside the plate.
    \item \textbf{GRASP} --- no instantaneous event: the gripper simply never closes on it.
  \end{itemize}
  \vspace{0.3em}
  \textbf{Timeout} --- a per-subtask step budget (\hl{only exists with the orchestrator}):
  \begin{itemize}
    \item A \hl{detection trigger} to react (retry / reset / redirect) --- \emph{not} a failure itself.
    \item Monotask has no timeout $\Rightarrow$ a failure is \hl{never caught mid-episode}.
  \end{itemize}
\end{frame}

% ===== 12. Orchestrator: three mechanisms (+ figure) =====
\begin{frame}{Orchestrator: a finite-state controller around the VLA}
  \begin{columns}[T]
    \begin{column}{0.56\textwidth}
      The VLA does \emph{only} GRASP / LIFT / PLACE; all checks use privileged state.
      \medskip

      Recovery at \hl{two levels} + one precondition:
      \begin{itemize}
        \item \hl{Level-1 --- local retry} \emph{(recovery)}: slip / timeout on LIFT or PLACE $\rightarrow$ re-GRASP the \textbf{same} orange.
        \item \hl{Level-2 --- target redirection} \emph{(recovery)}: GRASP timeout $\rightarrow$ switch to a \textbf{different} orange so the episode keeps scoring. \emph{Not} giving up --- task-level recovery.
        \item \hl{Home reset --- precondition} \emph{(not recovery)}: scripted move home before any target change.
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \includegraphics[width=\textwidth]{orchestrator_decision_flow.pdf}\\
      {\footnotesize Orchestrated decision flow.}
    \end{column}
  \end{columns}
\end{frame}

% ===== 13. Autonomous data generation =====
\begin{frame}{Probe: is the subtask policy just data-limited?}
  \begin{itemize}
    \item Run the orchestrator \hl{on its own}; keep only the \hl{successful} subtasks, discard the failed ones.
    \item \hl{Target redirection} salvages data from scenes that would otherwise be wasted.
    \item \hl{+414 Auto} episodes merged with 846 Teleop $\rightarrow$ \hl{1260} (\emph{Teleop+Auto}).
    \item Auto episodes are \hl{shorter / less dispersed} --- drawn from the policy's own visited states.
  \end{itemize}
  \vfill
  {\footnotesize Question: does adding in-distribution successful data improve the orchestrated policy?}
\end{frame}

% ===== 16. Results: obedience =====
\begin{frame}{Results: does it grasp the orange it was told to?}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{itemize}
        \item $\sim$\hl{3 in 4} grasps obey; better as the scene clears.
        \item Misgrabs are \hl{systematic}: the gripper closes on the orange just \emph{inward}.
          \begin{itemize}\item ``right'' (the \emph{most common} label) is the \hl{worst} obeyed.\end{itemize}
        \item More data $\neq$ better obedience.
        \item[$\Rightarrow$] A \hl{grounding} gap, not a \emph{coverage} gap.
      \end{itemize}

% ===== 17. Discussion & limitations =====
\begin{frame}{Discussion \& limitations}
  \begin{itemize}
    \item Decomposition creates \hl{real intervention points} --- failures are caught, not ignored.
    \item But it \hl{cannot fix a weak GRASP policy} --- the bottleneck, and worse as the scene fills.
    \item Auto data did not help $\Rightarrow$ the bottleneck is grasp \hl{quality}, not data \hl{volume}.
    \item \hl{Coverage gap}: trained on clean isolated subtasks, never on messy full runs.
    \item Success checks use \hl{privileged state} --- real-world deployment needs perception.
  \end{itemize}
\end{frame}

% ===== 18. Conclusion =====
\begin{frame}{Conclusion \& future work}
  \begin{itemize}
    \item Recovery is an \hl{execution-level} problem, not just a policy choice.
    \item The orchestrator enables \hl{retry / reset / redirection} --- and makes recovery \hl{measurable} at two levels.
    \item Orchestration alone $\neq$ robustness while the grasp policy is weak (a \hl{useful negative result}).
    \item \textbf{Next:} better GRASP / PLACE in late scenes; collect \hl{full, imperfect runs} --- \emph{before} more autonomous data.
  \end{itemize}
\end{frame}

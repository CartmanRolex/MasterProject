% ===== 2. Problem =====
\begin{frame}{The problem: long-horizon manipulation}
  \begin{itemize}
    \item Success needs \hl{grasp $\rightarrow$ lift $\rightarrow$ place} --- in order.
    \item Imitation policies learn from \hl{successful demos} $\Rightarrow$ brittle \hl{out-of-distribution} after any slip.
    \item A language interface can't \hl{detect} or \hl{interrupt} a failure.
  \end{itemize}
  \vfill
  \begin{block}{Angle}
    Fix the \hl{structure around the policy}, not just the policy.
  \end{block}
\end{frame}

% ===== 3. Idea + contributions =====
\begin{frame}{Idea: decompose, then orchestrate}
  \textbf{Hypothesis.} Robustness comes from \hl{language-conditioned subtasks} + an \hl{orchestrator} that checks each outcome and \hl{recovers}.
  \vfill
  \textbf{Contributions.}
  \begin{enumerate}
    \item Monotask baseline (\hl{ACT vs.\ SmolVLA}) --- policy class alone is not enough.
    \item \hl{Subtask dataset} + orchestrator: \hl{retry / reset / redirect}.
    \item \hl{Autonomous data} generation --- does more data help? (\emph{not clear-cut}).
  \end{enumerate}
\end{frame}

% ===== 6. Decompose =====
\begin{frame}{Decompose: GRASP $\rightarrow$ LIFT $\rightarrow$ PLACE}
  \begin{itemize}
    \item Each subtask = an explicit \hl{checkpoint}; \hl{verify the physical outcome} before the next step.
    \item GRASP targets \hl{one orange by position} (left / middle / right).
    \item \hl{Confirmed-grasp gate}: lift only on a secure grasp, not just a closed gripper.
  \end{itemize}

% ===== 7. Orchestrator =====
\begin{frame}{Orchestrator: a controller around the VLA}
  \begin{columns}[c]
    \begin{column}{0.54\textwidth}
      \begin{itemize}
        \item VLA does \hl{only} GRASP / LIFT / PLACE; checks use \hl{privileged state}.
        \item \hl{Timeout} = per-subtask budget $\rightarrow$ detection trigger.
        \item \hl{Level-1} retry --- \textbf{same} orange.
        \item \hl{Level-2} redirect --- \textbf{different} orange (task keeps scoring).
        \item \hl{Home reset} --- precondition: language switches target only when \hl{far from all oranges}.
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \centering
      \includegraphics[width=\textwidth]{orchestrator_decision_flow.pdf}
    \end{column}
  \end{columns}
\end{frame}

% ===== 10. Results: grasp bottleneck =====
\begin{frame}{Grasp acquisition is the bottleneck}
  \begin{itemize}
    \item Grasp secures an orange only \hl{$\sim$37\%} of attempts $\rightarrow$ \gn{55\%} under \hl{full} tuning.
    \item \hl{Lift and place already reliable} across every variant.
    \item Monotask lift jumps \hl{56\% $\rightarrow$ 94\%} with full tuning (no grasp gate).
  \end{itemize}
  \vfill
  \centering
  \includegraphics[width=\textwidth,height=0.60\textheight,keepaspectratio]{failure_modes_variants_subtask.pdf}
\end{frame}

% ===== 11. Results: recovery at matched stalls =====
\begin{frame}{Recovery, compared at matched stalls}
  {\footnotesize A \hl{stall} = one grasp budget stuck on a single orange (monotask intent read from \emph{proximity}). What happens next?}
  \vfill
  \centering\small
  \begin{tabular}{llccc}
    \toprule
    Training & Form. & Redirected & Placed after & \dots via \emph{different} orange \\
    \midrule
    Teleop      & Monotask          & 53.5\%     & 31.4\%        & 16.3\% \\
    Teleop      & \textbf{Subtask}  & \gn{100\%} & \gn{71.2\%}   & \gn{53.4\%} \\
    \midrule
    Teleop+Auto & Monotask          & 45.2\%     & 34.4\%        & 17.2\% \\
    Teleop+Auto & \textbf{Subtask}  & \gn{100\%} & \gn{43.2\%}   & \gn{33.7\%} \\
    \bottomrule
  \end{tabular}
  \vfill
  \begin{itemize}
    \item Orchestrator \hl{redirects every time} (Level-2 by design); monotask switches in \hl{$\sim$half}.
    \item The gap sits \hl{entirely in the different-orange column}; Level-1 stays marginal.
  \end{itemize}
\end{frame}

% ===== 13. Takeaways + future =====
\begin{frame}{Takeaways \& future work}
  \begin{itemize}
    \item Orchestration helps \hl{most where grasping is weak}: almost never ends empty-handed (Level-2 redirection).
    \item It \hl{cannot fix a weak grasp} --- a \hl{stronger policy} did (full fine-tuning, vision), not more structure.
    \item Two \hl{complementary levers}: better grasping $+$ graceful failure.
    \item This task shows the orchestration's \hl{floor}, not its ceiling.
  \end{itemize}
  \vfill
  \textbf{Next:} RL / targeted grasp demos \; $\cdot$ \; \hl{from-any-state} data \; $\cdot$ \; \hl{VLM success monitor}.
\end{frame}

% ----- B3: limitations -----
\begin{frame}{Backup: limitations}
  \begin{itemize}
    \item \hl{Privileged supervision} --- success read from simulator state; deployment needs perception.
    \item \hl{Matched-regime confound} --- full variant trains at batch 32 / 40k steps (controls bound it).
    \item \hl{Composite-label inconsistency} --- teleop labels by eye vs.\ scripted 3\,cm rule.
    \item \hl{Limited sampling} --- 100 seeded episodes per model.
    \item \hl{Slow execution} --- deliberately slow teleop $\rightarrow$ long rollouts.
  \end{itemize}
\end{frame}
